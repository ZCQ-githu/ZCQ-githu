// ==UserScript==
// @name         笔记本验机增强版脚本-炫彩快捷键版
// @namespace    https://zhaolegend.github.io/zcq
// @version      10.1
// @description  整合自动复制粘贴、质检提交、快捷键、动态按钮等功能
// @match        https://newocs.aihuishou.com/ts-ocs/inspection/operation*
// @match        https://newocs.aihuishou.com/inspection/reception
// @author       传奇
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    const storageKey = "copiedText";
    let copiedText = localStorage.getItem(storageKey) || '';

    // 统一配置参数
    const config = {
        interval: 500, // 统一延迟时间
        maxRetry: 3, // 最大重试次数
        debug: true, // 是否启用调试模式
        pulseSpeed: 1000, // 动态按钮脉冲速度
        keyBindings: {
            'ArrowDown': handleQualitySubmission, // 下箭头快捷键触发质检提交
        },
        gradient: {
            angle: 135, // 渐变角度
            variance: 100 // 渐变色相范围
        }
    };

    // 统一日志函数
    function log(message) {
        console.log(`[STEP] ${new Date().toLocaleTimeString()} - ${message}`);
    }

    // 通用函数
    async function delay(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }

    async function retryClickElement(tag, text, maxRetryTimes = config.maxRetry) {
        for (let attempt = 0; attempt < maxRetryTimes; attempt++) {
            const elements = Array.from(document.querySelectorAll(tag));
            const target = elements.find(el => el.textContent.includes(text));

            if (target) {
                log(`✅ 成功找到并点击元素: [${text}] 尝试次数: ${attempt + 1}`);
                target.click();
                return true;
            }

            log(`🔄 查找重试: [${text}] (${attempt + 1}/${maxRetryTimes})`);
            await delay(config.interval*2);
        }

        throw new Error(`❌ 重试失败，未找到元素: [${text}]`);
    }

    async function retryElementAction(selector, action, maxRetryTimes = config.maxRetry) {
        for (let attempt = 0; attempt < maxRetryTimes; attempt++) {
            const element = document.querySelector(selector);
            if (element) {
                action(element);
                return true;
            }
            log(`🔄 查找重试: [${selector}] (${attempt + 1}/${maxRetryTimes})`);
            await delay(config.interval*2);
        }
        throw new Error(`❌ 重试失败，未找到元素: [${selector}]`);
    }

    function handleException(error, context = "未知操作") {
        console.error(`❌ ${context} 发生错误:`, error.message);
        if (config.debug) console.trace();
    }

    function generateGradient() {
        const baseHue = Math.floor(Math.random() * 360);
        const color1 = `hsl(${baseHue}, 70%, 50%)`;
        const color2 = `hsl(${(baseHue + config.gradient.variance) % 360}, 85%, 45%)`;
        return `linear-gradient(${config.gradient.angle}deg, ${color1}, ${color2})`;
    }

    function generateShadow(color) {
        return `0 4px 15px ${color.replace(/hsl\((.*)\)/, 'hsla($1,0.3)')}`;
    }

    function createDynamicButton(text, onClick, rightOffset) {
        const btn = document.createElement('button');
        const gradient = generateGradient();
        const textColor = Math.random() > 0.5 ? '#fff' : '#000';

        btn.textContent = text;
        btn.style.cssText = `
            position: fixed;
            bottom: 25px;
            right: ${rightOffset}px;
            z-index: 9999;
            width: 150px;
            height: 60px;
            border-radius: 18px;
            background: ${gradient};
            color: ${textColor};
            border: none;
            cursor: pointer;
            font-weight: bold;
            font-size: 20px;
            box-shadow: ${generateShadow(gradient.match(/hsl\(.*?\)/)[0])};
            backdrop-filter: blur(4px);
            border: 2px solid rgba(255,255,255,0.2);
            transform-style: preserve-3d;
            perspective: 1000px;
            font-family:'Ma Shan Zheng', cursive !important;
        `;

        btn.classList.add('dynamic-button');
        btn.addEventListener('click', () => {
            btn.style.background = generateGradient();
            btn.style.boxShadow = generateShadow(gradient.match(/hsl\(.*?\)/)[0]);
            onClick();
        });

        document.body.appendChild(btn);
        return btn;
    }

    // 处理输入框粘贴和回车
    function handleInputWithStoredText(text) {
        const input = document.querySelector('input.ant-input[placeholder="请输入/扫码物品编码"]');
        if (input) {
          /*log(`找到输入框，开始粘贴内容: ${text}`);*/
            input.value = text;

            const inputEvent = new Event('input', { bubbles: true });
            input.dispatchEvent(inputEvent);

            const keydownEvent = new KeyboardEvent('keydown', {
                key: 'Enter',
                code: 'Enter',
                keyCode: 13,
                bubbles: true
            });
            input.dispatchEvent(keydownEvent);

            localStorage.removeItem(storageKey);
          /*log("已清空localStorage并触发回车事件");*/
        } else {
          /*log("输入框尚未出现，启动监听器等待");*/
            const inputObserver = new MutationObserver((mutations) => {
                mutations.forEach(mutation => {
                    Array.from(mutation.addedNodes).forEach(node => {
                        if (node.nodeType === Node.ELEMENT_NODE) {
                            const input = node.querySelector('input.ant-input[placeholder="请输入/扫码物品编码"]');
                            if (input) {
                              /*log("检测到输入框，开始粘贴内容");*/
                                handleInputWithStoredText(text);
                                inputObserver.disconnect();
                            }
                        }
                    });
                });
            });
            inputObserver.observe(document.body, { childList: true, subtree: true });
        }
    }

    // 监听div变化
    function setupDivObserver() {
        const observer = new MutationObserver((mutations) => {
            mutations.forEach(mutation => {
                Array.from(mutation.addedNodes).forEach(node => {
                    if (node.nodeType === Node.ELEMENT_NODE) {
                        const targetDiv = node.querySelector('.custom-title.has-prefix');
                        if (targetDiv) {
                            const text = targetDiv.textContent.trim();
                            if (text) {
                              /*log(`检测到新div元素，内容为: ${text}`);*/
                                localStorage.setItem(storageKey, text);
                              /*log("文本已存储到localStorage");*/
                                handleInputWithStoredText(text);
                            }
                        }
                    }
                });
            });
        });

        observer.observe(document.body, { childList: true, subtree: true });
    }

    // 一键质检逻辑
    async function handleQualitySubmission() {
        try {
            const requiredRadioButtons = Array.from(document.querySelectorAll("div.property-radio-item")).filter(el =>
                el.textContent.includes('键盘印刷字迹清晰，无磨损、掉漆') ||
                el.textContent.includes('键盘/触控板外观正常') ||
                el.textContent.includes('触摸板和按键外观完好') ||
                el.textContent.includes('机身无缝隙') ||
                el.textContent.includes('转轴/铰链功能正常') ||
                el.textContent.includes('不缺件') ||
                el.textContent.includes('部件之间缝隙正常') ||
                el.textContent.includes('无以上')
            );

            for (const radioButton of requiredRadioButtons) {
                radioButton.click();
                log(`✅ 点击单选框: ${radioButton.textContent}`);
            }
            await delay(config.interval);

            await retryClickElement("button", '点我刷新');
            await delay(config.interval);

            await retryClickElement("button", '提交');
            await delay(config.interval);

            await retryClickElement("span", '确 定');
            await delay(config.interval);

        } catch (error) {
            handleException(error, "质检操作处理失败");
        }
    }

    // 处理质检操作页面初始化
    function handleOperationPage() {
        if (config.debug) log('🔧 初始化质检操作页面...');
        setupDivObserver();
        createDynamicButton('✅一键质检', handleQualitySubmission, 90);
    }

    // 初始化脚本
    (function init() {
        // 动态加载 Google Fonts
        const link = document.createElement('link');
        link.href = "https://fonts.googleapis.com/css2?family=Ma+Shan+Zheng&display=swap";
        link.rel = 'stylesheet';
        document.head.appendChild(link);

        if (config.debug) log('🎮 炫彩快捷键增强版初始化中...');

        // 在质检接收页面加载后自动跳转到质检操作页面
        if (window.location.href.includes('inspection/reception') ||
            window.location.href.includes('inspection/receive')) {
            (async () => {
                try {
                    await delay(config.interval);
                    if (config.debug) log(`⏳ 等待 ${config.interval} 毫秒后跳转到质检操作页面`);
                    window.location.href = '/ts-ocs/inspection/operation';
                } catch (error) {
                    handleException(error, "自动跳转失败");
                }
            })();
        } else {
            // 在质检操作页面执行初始化操作
            handleOperationPage();
        }

        // 初始化全局快捷键绑定
        window.addEventListener('keydown', (e) => {
            const key = e.key;
            if (config.keyBindings[key]
                && !e.ctrlKey && !e.altKey && !e.metaKey && !e.shiftKey) {
                e.preventDefault();
                config.keyBindings[key]();
                if (config.debug) log(`⌨️ 快捷键触发: ${key}`);
            }
        }, true);

        // 启动div监听
        setupDivObserver();

        // 在 DOM 加载时执行
        document.addEventListener('DOMContentLoaded', () => {
            const storedText = localStorage.getItem(storageKey);
            if (storedText) {
                handleInputWithStoredText(storedText);
            }
        });

        // 监听SPA路由变化
        window.addEventListener('hashchange', () => {
            const storedText = localStorage.getItem(storageKey);
            if (storedText) {
                handleInputWithStoredText(storedText);
            }
        });
        window.addEventListener('popstate', () => {
            const storedText = localStorage.getItem(storageKey);
            if (storedText) {
                handleInputWithStoredText(storedText);
            }
        });

        // 定期检查
        setInterval(() => {
            const storedText = localStorage.getItem(storageKey);
            if (storedText) {
                handleInputWithStoredText(storedText);
            }

            const existingDiv = document.querySelector('.custom-title.has-prefix');
            if (existingDiv) {
                const text = existingDiv.textContent.trim();
                if (text) {
                    localStorage.setItem(storageKey, text);
                  /*log("检测到已存在的div元素，内容已更新");*/
                }
            }
        }, config.interval*2);
    })();
})();

使用介绍：
该脚本为笔记本初检主观.拆修的质检模板专门定制，其他部门因质检模板不同无法使用啾！
安装方法：
1. 打开任意浏览器安装 Tampermonkey 扩展。
2. 点击 Tampermonkey 图标，选择"创建新脚本"。
3. 将上述代码粘贴到新脚本中并保存。
4. 打开浏览器扩展的开发者模式。
5. 打开目标网站，脚本会自动运行。

// ==UserScript==
// @name         修改表格字体颜色
// @namespace    http://tampermonkey.net/
// @version      1.0
// @match        https://newocs.aihuishou.com/ts-ocs/inspection/operation
// @grant        GM_addStyle
// @run-at       document-start
// @author       ZCQ-github
// ==/UserScript==

(function() {
    'use strict';

    // 设置全局样式
    const globalStyles = `
        /* 修改中间单选框的字体颜色为霓虹色并高亮显示 */
        .property-radio-item {
            color: #198625 !important;
            text-shadow: 0 0 0.1px #198625;
        }

        /* 修改表格字体颜色 */
        .ant-table td, .ant-table th {
            color: #198625 !important;
        }
    `;

    GM_addStyle(globalStyles);

    // 创建和插入Matrix Rain元素的函数
    function createMatrixRain() {
      /*console.log('MatrixRainElement creation skipped as it is not needed');*/
    }

    // 监视DOM变化
    const observer = new MutationObserver((mutations) => {
       /*console.log('DOM mutation detected');*/
        createMatrixRain();
    });

    // 开始观察
    observer.observe(document.documentElement, {
        childList: true,
        subtree: true
    });

    // 页面加载时初始化
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', createMatrixRain);
    } else {
        createMatrixRain();
    }

    // 确保在micro-app加载后也能显示
    window.addEventListener('load', () => {
        setTimeout(createMatrixRain, 1000);
    });
})();

JavaScript Tampermonkey 

// ==UserScript==
// @name         自动登录
// @namespace    登录助手
// @version      2.0
// @description  自动完成SSO登录
// @author       传奇
// @match        https://sso.aihuishou.com/cas/login*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    // 核心配置
    const config = {
        username: 'chuanqi.zhao', // 把''内的账号改成自己的账号
        password: 'ZCQwan0.', // 把''内的密码改成自己的密码
        position: {
            latitude: 31.7117,
            longitude: 119.9448,
            accuracy: 15001
        }
    };

    // 处理地理位置
    navigator.geolocation.getCurrentPosition = function(success) {
        success({
            coords: {
                ...config.position,
                altitude: null,
                altitudeAccuracy: null,
                heading: null,
                speed: null
            },
            timestamp: Date.now()
        });
    };

    // 登录函数
    function login() {
        // 填写表单
        document.querySelector('#username').value = config.username;
        document.querySelector('#password').value = config.password;

        // 触发input事件（修复了模板字符串的语法错误）
        ['username', 'password'].forEach(id => {
            document.querySelector(`#${id}`).dispatchEvent(new Event('input', { bubbles: true }));
        });

        // 点击登录按钮
        document.querySelector('button[type="submit"]').click();
    }

    // 页面加载完成后执行
    if (document.readyState === 'complete') {
        login();
    } else {
        window.addEventListener('load', login);
    }
})();
