// ==UserScript==
// @name         笔记本初检主观.拆修辅助验机增强版脚本-炫彩快捷键版啾
// @namespace    https://zhaolegend.github.io/zcq
// @version      5.0
// @description  提升稳定性，模块化分离，优化延迟与重试逻辑，增强日志输出啾
// @match        https://newocs.aihuishou.com/ts-ocs/inspection/operation
// @match        https://newocs.aihuishou.com/inspection/reception
// @author       传奇啾
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    // 配置参数啾
    const config = {
        intervals: {
            submit: 500, // 延迟时间啾
        },
        maxRetry: 3, // 最大重试次数啾
        debug: true, // 是否启用调试模式啾
        pulseSpeed: 1000, // 动态按钮脉冲速度啾
        keyBindings: {
            'ArrowRight': handleQualitySubmission, // 右箭头快捷键触发质检提交啾
        },
        gradient: {
            angle: 135, // 渐变角度啾
            variance: 100, // 渐变色相范围啾
        }
    };

    // 延迟函数啾
    async function delay(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }

    // 查找包含特定文本的元素并点击啾
    async function retryClickElement(tag, text, maxRetryTimes = config.maxRetry) {
        for (let attempt = 0; attempt < maxRetryTimes; attempt++) {
            const elements = Array.from(document.querySelectorAll(tag));
            const target = elements.find(el => el.textContent.includes(text));

            if (target) {
                if (config.debug) console.log(`✅ 成功找到并点击元素: [${text}] 尝试次数: ${attempt + 1}啾`);
                target.click();
                return true;
            }

            if (config.debug) console.log(`🔄 查找重试: [${text}] (${attempt + 1}/${maxRetryTimes})啾`);
            await delay(config.intervals.submit);
        }

        throw new Error(`❌ 重试失败，未找到元素: [${text}]啾`);
    }

    // 日志与异常处理函数啾
    function handleException(error, context = "未知操作") {
        console.error(`❌ ${context} 发生错误:`, error.message);
        if (config.debug) console.trace();
    }

    // 生成随机渐变颜色啾
    function generateGradient() {
        const baseHue = Math.floor(Math.random() * 360);
        const color1 = `hsl(${baseHue}, 70%, 50%)`;
        const color2 = `hsl(${(baseHue + config.gradient.variance) % 360}, 85%, 45%)`;
        return `linear-gradient(${config.gradient.angle}deg, ${color1}, ${color2})`;
    }

    // 生成动态阴影颜色啾
    function generateShadow(color) {
        return `0 4px 15px ${color.replace(/hsl\((.*)\)/, 'hsla($1,0.3)')}`;
    }

    // 创建动态按钮啾
    function createDynamicButton(text, onClick, rightOffset) {
        const btn = document.createElement('button');
        const gradient = generateGradient();
        const textColor = Math.random() > 0.5 ? '#fff' : '#000';

        // 设置按钮基本样式啾
        btn.textContent = text;
        btn.style.cssText = `
            position: fixed;
            bottom: 25px;
            right: ${rightOffset}px;
            z-index: 9999;
            width: 150px;
            height: 60px;
            border-radius: 18px;
            background: ${gradient};
            color: ${textColor};
            border: none;
            cursor: pointer;
            font-weight: bold;
            font-size: 20px;
            box-shadow: ${generateShadow(gradient.match(/hsl\(.*?\)/)[0])};
            backdrop-filter: blur(4px);
            border: 2px solid rgba(255,255,255,0.2);
            transform-style: preserve-3d;
            perspective: 1000px;
            font-family:'Ma Shan Zheng', cursive !important; /* 添加手写体字体 */
        `;

        // 添加动态类啾
        btn.classList.add('dynamic-button');

        // 点击事件啾
        btn.addEventListener('click', () => {
            btn.style.background = generateGradient();
            btn.style.boxShadow = generateShadow(gradient.match(/hsl\(.*?\)/)[0]);
            onClick();
        });

        document.body.appendChild(btn);
        return btn;
    }

    /** 核心逻辑函数 **/
    // 一键质检逻辑啾
    async function handleQualitySubmission() {
        try {
            // 点击符合条件的单选框啾
            const requiredRadioButtons = Array.from(document.querySelectorAll("div.property-radio-item")).filter(el =>
                el.textContent.includes('键盘印刷字迹清晰，无磨损、掉漆') || el.textContent.includes('键盘/触控板外观正常') ||
                el.textContent.includes('触摸板和按键外观完好') || el.textContent.includes('机身无缝隙') ||
                el.textContent.includes('转轴/铰链功能正常') || el.textContent.includes('不缺件') ||
                el.textContent.includes('部件之间缝隙正常') ||
                el.textContent.includes('无以上')
            );

            for (const radioButton of requiredRadioButtons) {
                radioButton.click();
                if (config.debug) {
                    console.log(`✅ 点击单选框: ${radioButton.textContent}啾`);
                }
            }
            await delay(config.intervals.submit);

            // 点击 "点我刷新" 啾
            await retryClickElement("button", '点我刷新');
            await delay(config.intervals.submit);

            // 点击"提交"啾
            await retryClickElement("button", '提交');
            await delay(config.intervals.submit);

            // 点击 "确定"啾
            await retryClickElement("span", '确 定');

        } catch (error) {
            handleException(error, "质检操作处理失败啾");
        }
    }

    // 初始化脚本啾
    (function init() {
        // 动态加载 Google Fonts
        const link = document.createElement('link');
        link.href ="https://fonts.googleapis.com/css2?family=Ma+Shan+Zheng&display=swap";
        link.rel = 'stylesheet';
        document.head.appendChild(link);

        if (config.debug) console.log('🎮 炫彩快捷键增强版初始化中啾啾啾...');
        createDynamicButton('✅一键质检啾', handleQualitySubmission, 90);

        // 初始化全局快捷键绑定啾
        window.addEventListener('keydown', (e) => {
            const key = e.key;
            if (config.keyBindings[key]
                && !e.ctrlKey && !e.altKey && !e.metaKey && !e.shiftKey) {
                e.preventDefault();
                config.keyBindings[key]();
                if (config.debug) console.log(`⌨️ 快捷键触发: ${key}啾`);
            }
        }, true);
    })();

    // 在质检接收页面加载后自动跳转到质检操作页面啾
    if (window.location.href.includes('inspection/reception') || window.location.href.includes('inspection/receive')) {
        (async () => {
            try {
                // 使用 config.intervals.submit 变量作为等待时间
                await delay(config.intervals.submit);
                if (config.debug) console.log(`⏳ 等待 ${config.intervals.submit} 毫秒后跳转到质检操作页面啾`);
                window.location.href = '/ts-ocs/inspection/operation';
            } catch (error) {
                handleException(error, "自动跳转失败");
            }
        })();
    }
})();

使用介绍：
该脚本为笔记本初检主观.拆修的质检模板专门定制，其他部门因质检模板不同无法使用啾！
安装方法：
1. 打开任意浏览器安装 Tampermonkey 扩展。
2. 点击 Tampermonkey 图标，选择"创建新脚本"。
3. 将上述代码粘贴到新脚本中并保存。
4. 打开浏览器扩展的开发者模式。
5. 打开目标网站，脚本会自动运行。

// ==UserScript==
// @name         代码雨运营系统前端网页样式覆盖
// @namespace    http://tampermonkey.net/
// @version      2.0
// @match        https://newocs.aihuishou.com/ts-ocs/inspection/operation
// @grant        GM_addStyle
// @run-at       document-start
// @author       ZCQ-githu
// ==/UserScript==

(function() {
    'use strict';

    // 定义自定义元素
    class MatrixRainElement extends HTMLElement {
        constructor() {
            super();

            // 创建 Shadow DOM
            this.attachShadow({ mode: 'open' });

            // 创建样式
            const style = document.createElement('style');
            style.textContent = `
                :host {
                    position: fixed;
                    top: 0;
                    left: 0;
                    width: 100vw;
                    height: 100vh;
                    pointer-events: none;
                    z-index: 9999; /* Ensure it's on top */
                }
                canvas {
                    position: fixed;
                    top: 0;
                    left: 0;
                    width: 100vw;
                    height: 100vh;
                    z-index: 9999; /* Ensure it's on top */
                    opacity: 0.2; /* Increased transparency */
                }
            `;

            // 创建 canvas
            this.canvas = document.createElement('canvas');
            this.ctx = this.canvas.getContext('2d');

            // 添加到 Shadow DOM
            this.shadowRoot.appendChild(style);
            this.shadowRoot.appendChild(this.canvas);

            // 初始化属性
            this.fontSize = 14;
            this.chars = Array.from('ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789@#$%^&*');
            this.drops = [];

            // 绑定方法
            this.animate = this.animate.bind(this);
            this.handleResize = this.handleResize.bind(this);

            // 添加事件监听
            window.addEventListener('resize', this.handleResize);

            // 初始化
            this.initializeCanvas();
            this.startAnimation();
        }

        initializeCanvas() {
            // 设置 canvas 尺寸
            this.canvas.width = window.innerWidth;
            this.canvas.height = window.innerHeight;
            this.columns = Math.floor(this.canvas.width / this.fontSize);
            this.drops = new Array(this.columns).fill(1);
            console.log(`Canvas initialized: width=${this.canvas.width}, height=${this.canvas.height}`);
        }

        handleResize() {
            this.initializeCanvas();
        }

        draw() {
            // 创建渐变效果的背景
            this.ctx.fillStyle = 'rgba(0, 0, 0, 0.02)';
            this.ctx.fillRect(0, 0, this.canvas.width, this.canvas.height);

            // 设置文字样式
            this.ctx.font = `${this.fontSize}px monospace`;

            // 绘制字符
            for (let i = 0; i < this.drops.length; i++) {
                const char = this.chars[Math.floor(Math.random() * this.chars.length)];
                const x = i * this.fontSize;
                const y = this.drops[i] * this.fontSize;

                //透明度范围
                const opacity = Math.random() * (0.8 - 0.3) + 0.3;
                this.ctx.fillStyle = `rgba(0, 255, 0, ${opacity})`;
                this.ctx.fillText(char, x, y);

                // 更新位置
                if (y > this.canvas.height && Math.random() > 0.975) {
                    this.drops[i] = 0;
                }
                this.drops[i]++;
            }
        }

        animate() {
            this.draw();
            this.animationId = requestAnimationFrame(this.animate);
            console.log('Animation frame requested', this.animationId);
        }

        startAnimation() {
            this.animate();
        }

        // 当元素被移除时清理
        disconnectedCallback() {
            window.removeEventListener('resize', this.handleResize);
            if (this.animationId) {
                cancelAnimationFrame(this.animationId);
            }
        }
    }

    // 注册自定义元素
    customElements.define('matrix-rain', MatrixRainElement);
    console.log('MatrixRainElement registered');

    // 设置全局样式
    const globalStyles = `
        /* 保护现有UI元素 */
        .global-header,
        .ant-layout-sider,
        .ant-menu,
        .ant-form-item,
        .ant-radio-wrapper,
        .ant-checkbox-wrapper,
        .ant-select,
        .ant-input,
        .ant-btn,
        .ant-table,
        micro-app {
            position: relative;
            z-index: 1;
        }

        /* 修改中间单选框的字体颜色为霓虹色并高亮显示 */
        .property-radio-item {
            color: #00BFFF!important;
          /*text-shadow: 0 0 10px #ff00ff, 0 0 20px #ff00ff, 0 0 30px #ff00ff;*/
        }
    `;

    GM_addStyle(globalStyles);

    // 创建和插入Matrix Rain元素的函数
    function createMatrixRain() {
        let matrixRain = document.querySelector('matrix-rain');
        if (!matrixRain) {
            matrixRain = document.createElement('matrix-rain');
            document.body.insertBefore(matrixRain, document.body.firstChild);
            console.log('MatrixRainElement created and inserted');
        } else {
            console.log('MatrixRainElement already exists');
        }
    }

    // 监视DOM变化
    const observer = new MutationObserver((mutations) => {
        console.log('DOM mutation detected');
        createMatrixRain();
    });

    // 开始观察
    observer.observe(document.documentElement, {
        childList: true,
        subtree: true
    });

    // 页面加载时初始化
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', createMatrixRain);
    } else {
        createMatrixRain();
    }

    // 确保在micro-app加载后也能显示
    window.addEventListener('load', () => {
        setTimeout(createMatrixRain, 1000);
    });
})();

// ==UserScript==
// @name         自动登录
// @namespace    登录助手
// @version      2.0
// @description  自动完成SSO登录
// @author       传奇
// @match        https://sso.aihuishou.com/cas/login*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    // 核心配置
    const config = {
        username: 'chuanqi.zhao', // 把''内的账号改成自己的账号
        password: 'ZCQwan0.', // 把''内的密码改成自己的密码
        position: {
            latitude: 31.7117,
            longitude: 119.9448,
            accuracy: 15001
        }
    };

    // 处理地理位置
    navigator.geolocation.getCurrentPosition = function(success) {
        success({
            coords: {
                ...config.position,
                altitude: null,
                altitudeAccuracy: null,
                heading: null,
                speed: null
            },
            timestamp: Date.now()
        });
    };

    // 登录函数
    function login() {
        // 填写表单
        document.querySelector('#username').value = config.username;
        document.querySelector('#password').value = config.password;

        // 触发input事件（修复了模板字符串的语法错误）
        ['username', 'password'].forEach(id => {
            document.querySelector(`#${id}`).dispatchEvent(new Event('input', { bubbles: true }));
        });

        // 点击登录按钮
        document.querySelector('button[type="submit"]').click();
    }

    // 页面加载完成后执行
    if (document.readyState === 'complete') {
        login();
    } else {
        window.addEventListener('load', login);
    }
})();
